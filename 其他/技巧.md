点击遮罩层后关闭，会遇到事件冒泡，解决方法是检查事件的触发对象和绑定对象是否一致
```js
$('.overlay').click(function(e){
    // e.targetb绑定的是点击时触发的dom节点
    // e.currentTarget绑定的click事件的dom节点
    // 如果不一致，说明点击的不是期望值
    if (e.target == e.currentTarget)
    closeOverlay();
});
